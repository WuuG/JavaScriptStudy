# Set
实现了iterator接口，可以遍历（扩展运算符，for...of）,无重复
``` javascript
  //可接受可迭代数据,会自动去重
  let s2 = new Set(["wuug", "wis", "lwx", "wuug"]);
  console.log(s2, typeof s2); //object类型
  //元素个数
  console.log(s2.size);
  //添加新元素
  s2.add("wug");
  //删除元素
  s2.delete("wis");
  //检测元素
  console.log(s2.has("lwx"));
  //清空集合
  // s2.clear();
  //遍历
  for (let v of s2) {
    console.log(v);
  }
```
# Map
键值对的集合， 键不仅仅是字符串，还可能使其他类型的值。

实现了iterator接口，可以遍历
``` javascript
let m = new Map();
m.set("name", "wuug");
m.set("jump", function () {
  console.log("跳起来了！！");
});
let key = {
  school: "fzu",
};
//添加
m.set(key, ["a", "b", "c"]);
//大小
console.log(m.size);
//删除
// m.delete(key);
//获取
m.get("jump")();
//清空
// m.clear();
for (const v of m) { //这里的const 还是挺有意思的，v是一个指针对象哦
  console.log(v);
}
console.log(m, typeof m); //object
```
# class类
对象的模板,对象的原型的更清晰的写法
``` javascript
// ES5写法
function Phone(brand, price) {
  this.brand = brand;
  this.price = price;
}
Phone.prototype.call = function () {
  console.log("这个是" + this.brand + "的手机");
};
let Huawei = new Phone("华为", 2922);
Huawei.call();
console.log(Huawei);
//ES6 class写法
class Phone2 {
  //构造函数  一旦new就会调用这个方法
  constructor(brand, price) {
    this.brand = brand;
    this.price = price;
  }
  //class里的函数写法 一定要这样写哦
  call() {
    console.log("这个是" + this.brand + "的手机");
  }
}
let oppo = new Phone2("oppo", 2923);
oppo.call();
console.log(oppo);
```
## 类的静态成员
``` javascript
function Phone() {}
//函数对象添加了属性和方法，实例对象是无法访问的，二者并不是相通的
Phone.brand = "xiaoMi";
Phone.slogen = function () {
  console.log(this.brand + ":干翻华为！！");
};
Phone.prototype.size = "12inch";
let xiaomi = new Phone();
console.log(xiaomi.brand); //undefined
console.log(xiaomi.size);
console.log(Phone.brand); //xiaomi
Phone.slogen(); //xiaomi：干翻华为
class Phone2 {
  //静态成员属于Phone2，而不属于实例对象
  static name = "手机";
  static change() {
    console.log(this.name + "发生了改变！");
  }
}
let huaWei = new Phone2();
console.log(huaWei.name); //undefined
console.log(Phone2.name); //手机
```
## 类继承
### ES5类继承
``` javascript
//ES5的继承
function Phone(brand, price) {
  this.brand = brand;
  this.price = price;
}
Phone.prototype.add = function () {
  console.log(this.brand + "添加了加个为" + this.price);
};
//智能手机
function SmartPhone(brand, price, screen, memory) {
  //this指向SmartP的实例对象，构造函数继承传参,构造函数的继承,这里的call是固定的
  Phone.call(this, brand, price);
  this.screen = screen;
  this.memory = memory;
}
//设置子级构造函数的原型,在原型上进行父级方法的继承
SmartPhone.prototype = new Phone();
SmartPhone.prototype.constructor = SmartPhone; //矫正constructor函数

//给子类对象添加方法
SmartPhone.prototype.photo = function () {
  console.log("智能手机拍照");
};
const mix = new SmartPhone("xiaomi", 2092, "16ich", "12g");
console.log(mix);
mix.add();
mix.photo();
```
### ES6类的继承
``` javascript
class Phone {
  constructor(brand, price) {
    this.brand = brand;
    this.price = price;
  }
  call() {
    console.log(this.brand + "可以打电话啦");
  }
}
class SmartPhone extends Phone {
  constructor(brand, price, color, size) {
    //父类的构造函数的继承
    super(brand, price);
    this.color = color;
    this.size = size;
  }
  photo() {
    console.log("拍照");
  }
}
const xiaomi = new SmartPhone("xiaomi", 2032, "red", "20ich");
console.log(xiaomi);
xiaomi.call();
xiaomi.photo();
```
## 子类对父类方法的重写
``` javascript
class Phone {
  constructor(brand, price) {
    this.brand = brand;
    this.price = price;
  }
  call() {
    console.log("这是父类的call");
  }
}
class SmartPhone extends Phone {
  constructor(brand, price, color, size) {
    //父类的构造函数的继承,这个super是一定要的
    super(brand, price);
    this.color = color;
    this.size = size;
  }
  call() {
    //子类不能够调用父类的同名方法的
    // this.call();
    console.log("这是子类的call");
  }
}
const xiaomi = new SmartPhone("xiaomi", 2032, "red", "20ich");
console.log(xiaomi);
xiaomi.call();
```
## get和set函数
``` javascript
//1.get
class Phone {
  //get 可用于返回一些需要计算的值，在计算后返回
  get price() {
    console.log("读取价格");
    return "$1290";
  }
  //set 函数是一定要有参数的,可用于类型检测
  set price(newValue) {
    // return; //只要return就不会对数据进行修改
    console.log("修改价格属性");
  }
}
let s = new Phone();
console.log(s.price); //这里s.price的值为 return的值 这很类似vue中的computerd
s.price = "$10";
console.log(s.price);
```
